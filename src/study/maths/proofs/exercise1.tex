\begin{problem}
  Explain the error in the following "proof" that $2 = 1$. \\
  Let $x = y$. Then,

\begin{align}
x^2 &= xy \\
x^2 - y^2 &= xy - y^2 \\
(x+y)(x-y) &= y(x-y) \\
x+y &= y \\
2y &= y \\
2 &= 1
\end{align}
\end{problem}

\begin{solution}
  Since $x = y$, $x-y = 0$ and therefore, we cannot divided by $x-y$ in step $3$ to get $x+y = y$ from $(x+y)(x-y) = y(x-y)$. Thus, solved.
\end{solution}
